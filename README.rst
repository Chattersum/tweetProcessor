Add the following to your bashrc or activate script

export MONGO_PORT=
export MONGO_USER=
export MONGO_PASS=
export MONGO_DB=

export TWITTER_CONSUMER_KEY=
export TWITTER_CONSUMER_SECRET=
export TWITTER_ACCESS_TOKEN=
export TWITTER_ACCESS_TOKEN_SECRET=
